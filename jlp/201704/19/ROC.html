files <- grep("*csv", dir("../", full.names = TRUE), value=TRUE)
names <- file_path_sans_ext(grep("*csv", dir("../"), value=TRUE))

dat <- lapply(files, read.csv, header=TRUE)

d0 <- lapply(dat, 
    function(x){
        d0 <- x[, -1]
        Xs <- d0[,c(1:2)]
        Ys <- d0[,-c(1:2)]
        dd <- data.frame(Xs, melt(Ys, measure.vars= names(Ys)))
        dout <- dd[complete.cases(dd),]
        return(dout)
    })

names(d0) <- names
DenPro <- lapply(d0, 
      function(d){
          ggplot(d, aes(x = Protein.ELISA, group = as.factor(value), col =as.factor(value))) + 
          #geom_histogram(position = 'identity', binwidth = 0.01, alpha = 0.2) + 
          geom_density() + 
          theme(legend.position = "none") +
          facet_grid(variable ~ .)
      })

DenCell <- lapply(d0, 
      function(d){
          ggplot(d, aes(x = Cell.ELISA, group = as.factor(value), col =as.factor(value))) + 
          #geom_histogram(position = 'identity',bins = 50, alpha = 0.2) +
          theme(legend.position = "none") +
          geom_density() + 
          facet_grid(variable ~ .)
      })
d1 <- lapply(d0, function(v){
               v[v$variable == "IHC",]
      })

d2 <- lapply(d0, function(v){
               v[v$variable == "IB",]
      })
rrP <- list()
for(i in 1:length(d1)){
rrP[[i]] <- roc(d1[[i]]$value ~ d1[[i]]$Protein.ELISA)
}


rrC <- list()
for(i in 1:length(d1)){
rrC[[i]] <- roc(d1[[i]]$value ~ d1[[i]]$Cell.ELISA)
}
